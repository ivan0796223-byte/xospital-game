@app.route("/register", methods=["GET", "POST"]) def register(): if request.method == "POST": try: login = request.form.get("login") password = request.form.get("password") if not login or not password: return "Заполни все поля" # проверка существующего пользователя if User.query.filter_by(login=login).first(): return "Пользователь уже есть" user = User(login=login, password=password) db.session.add(user) db.session.commit() return redirect("/login") except Exception as e: return f"Ошибка: {e}" # 👈 покажет причину return render_template("register.html")
